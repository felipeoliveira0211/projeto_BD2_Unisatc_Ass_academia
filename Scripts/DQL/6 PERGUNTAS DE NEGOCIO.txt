--Pergunta 1
Quais funcionários venderam mais assinaturas e qual a taxa de cancelamento das vendas feitas por eles?

Objetivo: avaliar desempenho comercial e qualidade das vendas.

Usa: JOIN, subconsulta, GROUP BY--

CREATE NONCLUSTERED INDEX IX_funcionarios_cargo
ON funcionarios (id_cargo)
INCLUDE (id, nome_completo);

CREATE NONCLUSTERED INDEX IX_cargos_id_nome
ON cargos (id)
INCLUDE (nome);

CREATE NONCLUSTERED INDEX IX_assinaturas_funcionario_status_renovacao
ON assinaturas (
    id_funcionario_venda,
    status_assinatura,
    renovacao_automatica
);

SELECT
    f.id AS id_funcionario,
    f.nome_completo AS funcionario,
    c.nome AS cargo,
    COUNT(ass.id) AS total_vendas,
    SUM(CASE WHEN ass.status_assinatura = 'I' THEN 1 ELSE 0 END) AS vendas_canceladas,
    CAST(
        100.0 * SUM(CASE WHEN ass.status_assinatura = 'I' THEN 1 ELSE 0 END)
        / COUNT(ass.id)
        AS DECIMAL(5,2)
    ) AS percentual_cancelamento,
    (
        SELECT COUNT(*)
        FROM assinaturas ass2
        WHERE ass2.id_funcionario_venda = f.id
          AND ass2.renovacao_automatica = 1
    ) AS vendas_com_renovacao_automatica
FROM funcionarios f
INNER JOIN cargos c
    ON c.id = f.id_cargo
INNER JOIN assinaturas ass
    ON ass.id_funcionario_venda = f.id
GROUP BY
    f.id,
    f.nome_completo,
    c.nome
ORDER BY total_vendas DESC;

--Pergunta 2
Quais planos geram maior faturamento e qual o percentual de inadimplência por plano?

Objetivo: comparar planos por receita e risco financeiro.

Usa: JOIN, agregação, CASE--

CREATE NONCLUSTERED INDEX IX_assinaturas_plano_id
ON assinaturas (id_plano)
INCLUDE (id);

CREATE NONCLUSTERED INDEX IX_faturas_assinatura_status_valores
ON faturas_pagamento (id_assinatura, status_pagamento)
INCLUDE (valor_pago, valor_fatura, multa_aplicada);

CREATE NONCLUSTERED INDEX IX_faturas_assinatura_status
ON faturas_pagamento (
    id_assinatura,
    status_pagamento
);

SELECT
    p.nome AS plano,
    COUNT(DISTINCT ass.id) AS total_assinaturas,
    SUM(CASE 
            WHEN fp.status_pagamento = 'C' THEN fp.valor_pago 
            ELSE 0 
        END) AS valor_recebido,
    SUM(CASE 
            WHEN fp.status_pagamento = 'P' THEN fp.valor_fatura + fp.multa_aplicada 
            ELSE 0 
        END) AS valor_pendente,
    CAST(
        100.0 * SUM(CASE WHEN fp.status_pagamento = 'P' THEN 1 ELSE 0 END)
        / COUNT(fp.id)
        AS DECIMAL(5,2)
    ) AS percentual_faturas_pendentes
FROM planos p
INNER JOIN assinaturas ass
    ON ass.id_plano = p.id
INNER JOIN faturas_pagamento fp
    ON fp.id_assinatura = ass.id
GROUP BY p.nome
ORDER BY valor_recebido DESC;

--Pergunta 3
Quais são os alunos mais frequentes no mês e qual posição eles ocupam no ranking?

Objetivo: identificar alunos mais engajados.

Usa: CTE, JOIN, função de janela RANK--

CREATE NONCLUSTERED INDEX IX_assinaturas_aluno_status_plano
ON assinaturas (id_aluno, status_assinatura)
INCLUDE (id_plano);

CREATE NONCLUSTERED INDEX IX_planos_id_nome
ON planos (id)
INCLUDE (nome);

CREATE NONCLUSTERED INDEX IX_acessos_aluno_data_tipo
ON acessos_catraca (
    id_aluno,
    data_hora_entrada,
    sentido_acesso,
    tipo_acesso
);

WITH frequencia_mensal AS (
    SELECT
        a.id AS id_aluno,
        a.nome_completo,
        p.nome AS plano,
        COUNT(ac.id) AS total_acessos
    FROM alunos a
    INNER JOIN assinaturas ass
        ON ass.id_aluno = a.id
    INNER JOIN planos p
        ON p.id = ass.id_plano
    INNER JOIN acessos_catraca ac
        ON ac.id_aluno = a.id
    WHERE ac.sentido_acesso = 'E'
      AND ac.tipo_acesso = 'A'
      AND MONTH(ac.data_hora_entrada) = 5
      AND YEAR(ac.data_hora_entrada) = 2026
      AND ass.status_assinatura = 'A'
    GROUP BY
        a.id,
        a.nome_completo,
        p.nome
)
SELECT
    id_aluno,
    nome_completo,
    plano,
    total_acessos,
    RANK() OVER (ORDER BY total_acessos DESC) AS posicao_ranking
FROM frequencia_mensal
ORDER BY posicao_ranking;

--Pergunta 4
Quais alunos ativos possuem faturas pendentes vencidas e qual o valor total em atraso?

Objetivo: identificar alunos inadimplentes que podem ter acesso bloqueado.

Usa: CTE, JOIN, GROUP BY, função--

WITH faturas_vencidas AS (
    SELECT
        fp.id_assinatura,
        COUNT(fp.id) AS qtd_faturas_vencidas,
        SUM(fp.valor_fatura + fp.multa_aplicada) AS valor_total_em_atraso,
        MIN(fp.data_vencimento) AS vencimento_mais_antigo
    FROM faturas_pagamento fp
    WHERE fp.status_pagamento = 'P'
      AND fp.data_vencimento < CAST(GETDATE() AS DATE)
    GROUP BY fp.id_assinatura
)
SELECT
    a.id AS id_aluno,
    a.nome_completo,
    dbo.fn_calcular_idade(a.data_nascimento) AS idade,
    p.nome AS plano,
    fv.qtd_faturas_vencidas,
    fv.valor_total_em_atraso,
    fv.vencimento_mais_antigo
FROM faturas_vencidas fv
INNER JOIN assinaturas ass
    ON ass.id = fv.id_assinatura
INNER JOIN alunos a
    ON a.id = ass.id_aluno
INNER JOIN planos p
    ON p.id = ass.id_plano
WHERE a.ativo = 1
  AND ass.status_assinatura = 'A'
ORDER BY fv.valor_total_em_atraso DESC;

--Pergunta 5
Quais dias da semana e horários possuem maior movimentação na catraca?

Objetivo: entender horários de pico da academia.

Usa: CTE, GROUP BY, DATEPART--

WITH movimento_horario AS (
    SELECT
        DATENAME(WEEKDAY, data_hora_entrada) AS dia_semana,
        DATEPART(HOUR, data_hora_entrada) AS hora,
        COUNT(*) AS total_acessos,
        SUM(CASE WHEN tipo_acesso = 'A' THEN 1 ELSE 0 END) AS acessos_alunos,
        SUM(CASE WHEN tipo_acesso = 'V' THEN 1 ELSE 0 END) AS acessos_visitantes
    FROM acessos_catraca
    WHERE sentido_acesso = 'E'
    GROUP BY
        DATENAME(WEEKDAY, data_hora_entrada),
        DATEPART(HOUR, data_hora_entrada)
)
SELECT
    dia_semana,
    hora,
    total_acessos,
    acessos_alunos,
    acessos_visitantes,
    CAST(
        100.0 * acessos_visitantes / total_acessos
        AS DECIMAL(5,2)
    ) AS percentual_visitantes
FROM movimento_horario
ORDER BY total_acessos DESC;

INCLUDE (

--Pergunta 6
Quais alunos tiveram melhora na composição corporal entre a primeira e a última avaliação?

Objetivo: verificar evolução dos alunos acompanhados pela academia.

Usa: CTE, função de janela ROW_NUMBER, JOIN--

WITH avaliacoes_ordenadas AS (
    SELECT
        af.id_aluno,
        a.nome_completo,
        af.data_avaliacao,
        af.peso_kg,
        af.percentual_gordura,
        af.massa_magra,
        ROW_NUMBER() OVER (
            PARTITION BY af.id_aluno 
            ORDER BY af.data_avaliacao ASC
        ) AS primeira_avaliacao,
        ROW_NUMBER() OVER (
            PARTITION BY af.id_aluno 
            ORDER BY af.data_avaliacao DESC
        ) AS ultima_avaliacao
    FROM avaliacoes_fisicas af
    INNER JOIN alunos a
        ON a.id = af.id_aluno
)
SELECT
    primeira.nome_completo,
    primeira.data_avaliacao AS data_primeira_avaliacao,
    ultima.data_avaliacao AS data_ultima_avaliacao,
    primeira.percentual_gordura AS gordura_inicial,
    ultima.percentual_gordura AS gordura_atual,
    primeira.massa_magra AS massa_magra_inicial,
    ultima.massa_magra AS massa_magra_atual,
    primeira.percentual_gordura - ultima.percentual_gordura AS reducao_gordura,
    ultima.massa_magra - primeira.massa_magra AS ganho_massa_magra
FROM avaliacoes_ordenadas primeira
INNER JOIN avaliacoes_ordenadas ultima
    ON ultima.id_aluno = primeira.id_aluno
WHERE primeira.primeira_avaliacao = 1
  AND ultima.ultima_avaliacao = 1
  AND primeira.data_avaliacao <> ultima.data_avaliacao
ORDER BY reducao_gordura DESC;
