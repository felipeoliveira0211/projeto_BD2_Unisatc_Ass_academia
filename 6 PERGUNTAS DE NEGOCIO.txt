
/*PERGUNTAS/*
-- 1. Planos mais vendidos e receita
CREATE INDEX IX_ASSINATURAS_PLANO
ON assinaturas (id_plano);

-- 2. Ticket médio mensal
CREATE INDEX IX_FATURAS_ASSINATURA_STATUS
ON faturas_pagamento (id_assinatura, status_pagamento);

-- 3. Pico de horário catraca
CREATE INDEX IX_ACESSOS_DATA_HORA
ON acessos_catraca (data_hora_entrada, sentido_acesso);

-- 4. Novos alunos por mês
CREATE INDEX IX_ALUNOS_DATA_CADASTRO
ON alunos (data_cadastro);

-- 5. Funcionário com mais vendas
CREATE INDEX IX_ASSINATURAS_FUNCIONARIO
ON assinaturas (id_funcionario_venda);

-- 6. Receita total mensal
CREATE INDEX IX_FATURAS_STATUS_VALOR
ON faturas_pagamento (status_pagamento, valor_pago);

/*1. Quais são os planos mais vendidos e quais geram maior receita?/*
SELECT
    p.nome AS plano,
    COUNT(a.id) AS total_assinaturas,
    SUM(a.valor_acordado) AS receita_gerada
FROM planos p
INNER JOIN assinaturas a
    ON p.id = a.id_plano
GROUP BY p.nome
ORDER BY receita_gerada DESC;

/*2. Qual o ticket médio mensal de nossos clientes?/*
SELECT
    CAST(AVG(gasto_mensal) AS DECIMAL(10,2)) AS ticket_medio
FROM (
    SELECT
        a.id_aluno,
        SUM(f.valor_pago) AS gasto_mensal
    FROM faturas_pagamento f
    INNER JOIN assinaturas a
        ON a.id = f.id_assinatura
    WHERE f.status_pagamento = 'C'
    GROUP BY a.id_aluno
) AS x;
/*3. Qual é o pico de horário de acessos nas catracas?/*
SELECT
    DATEPART(HOUR, data_hora_entrada) AS hora,
    COUNT(*) AS quantidade_acessos
FROM acessos_catraca
WHERE sentido_acesso = 'E'
GROUP BY DATEPART(HOUR, data_hora_entrada)
ORDER BY quantidade_acessos DESC;

/*4. Quantos alunos novos foram captados por mês na academia?/*
SELECT
    YEAR(data_cadastro) AS ano,
    MONTH(data_cadastro) AS mes,
    COUNT(*) AS novos_alunos
FROM alunos
GROUP BY
    YEAR(data_cadastro),
    MONTH(data_cadastro)
ORDER BY ano, mes;

/*5. Qual funcionário mais vendeu assinaturas e qual a receita gerada?(utilizando function)/*

CREATE FUNCTION fn_vendas_funcionario (@id_funcionario INT)
RETURNS TABLE
AS
RETURN
(
    SELECT
        f.id AS id_funcionario,
        f.nome_completo,
        COUNT(a.id) AS total_vendas,
        SUM(a.valor_acordado) AS receita_total
    FROM funcionarios f
    INNER JOIN assinaturas a
        ON f.id = a.id_funcionario_venda
    WHERE f.id = @id_funcionario
    GROUP BY f.id, f.nome_completo
);
/*puxando a função/*
SELECT *
FROM dbo.fn_vendas_funcionario(1);

/*6. Qual a receita mensal total da academia?/*
SELECT
    SUM(valor_pago) AS receita_total_mes
FROM faturas_pagamento
WHERE status_pagamento = 'C';
